00058     /**
00059      * 페이징처리를 하기 위한 방법을 제공한다. 입력맵에 반드시 "_pageNo" 와 "_recordCountPerPage" 값이 있어야

00064     protected void setPagenatedParams(Map params) {
00065         String strPageNo = (String) params.get("nc_pageNo");
00066         int pageNo = 1;
00067         if (strPageNo != null && strPageNo.length() > 0) {
00068             pageNo = Integer.parseInt(strPageNo);
00069             if (pageNo == 0)
00070                 pageNo = 1;
00071         }
00072         String strRecordCountPerPage = (String) params
00073                 .get("nc_recordCountPerPage");
00074         int recordCountPerPage = 10;
00075         if (strRecordCountPerPage != null && strRecordCountPerPage.length() > 0) {
00076             recordCountPerPage = Integer.parseInt(strRecordCountPerPage);
00077         }
00078         params.put("nc_pageNo", String.valueOf(pageNo));
00079         params.put("nc_recordCountPerPage", String.valueOf(recordCountPerPage));
00080         params.put("nc_lastRowIndex", String.valueOf(pageNo
00081                 * recordCountPerPage));
00082         params.put("nc_firstRowIndex", String.valueOf((pageNo - 1)
00083                 * recordCountPerPage + 1));
00084     }

00086     /**
00087      * q_ 또는 p_로 시작하는 params를 q_ 또는 p_를 제외한 params로 변경한다.
00088      */
00089     protected Map<String, String> getRequestDataFieldMap(IDataSet requestData) {
00090 
00091         Map<String, String> reqFields = requestData.getFieldMap();
00092         Map<String, String> fields = new CaseIgnoreHashMap();
00093 
00094         Iterator<String> keyIter = reqFields.keySet().iterator();
00095 
00096         while (keyIter.hasNext()) {
00097             String key = (String) keyIter.next();
00098             String value = reqFields.get(key);
00099             String newKey;
00100             if (key != null
00101                     && key.length() >= 3
00102                     && (key.substring(0, 2).equalsIgnoreCase("q_") || key.substring(0, 2)
00103                             .equalsIgnoreCase("p_"))) {
00104                 newKey = key.substring(2);
00105                 fields.put(newKey, value);
00106             } else {
00107                 fields.put(key, value);
00108             }
00109         }
00110         return fields;
00111     }
